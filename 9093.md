`c++

#include<iostream>
#include<stack>
#include<string>

using namespace std;

int main(){
	ios_base::sync_with_stdio(false);
	cin.tie(nullptr);
	
	int t;
	//stack<char> s;
	string sentence;
	
	cin >> t;
	cin.ignore();
	while(t--){
		getline(cin, sentence);
		sentence += '\n'; //반복문 탈출 조건을 쉽게잡기위해서 뒤에 개행문자를 추가해놓고, 탈출 조건을 공백과 개행으로 잡는다.
		stack<char> s; //스택을 문장마다 하나씩 쓰는게 좋으니까 while문 안에다가 생성하자
		for(char ch: sentence){
			if(ch == ' ' || ch == '\n'){
				while(!s.empty()){
					cout << s.top();
					s.pop();
				}
				cout << ch;
			}
			else{
				s.push(ch);
			}
			
		}
	}
	return 0;
}
